<script>
(function initHistoryStack() {
    window.historyStack = [];
    window.historyIndex = -1;
    window.isHistoryNavigating = false;

    window.recordHistory = function() {
        if (window.isHistoryNavigating || (typeof editState === 'undefined' || !editState) || !editState.resume) return;
        const currentState = JSON.stringify(editState.resume);
        if (window.historyIndex >= 0 && window.historyStack[window.historyIndex] === currentState) return;

        if (window.historyIndex < window.historyStack.length - 1) {
            window.historyStack = window.historyStack.slice(0, window.historyIndex + 1);
        }

        window.historyStack.push(currentState);
        if (window.historyStack.length > 50) {
            window.historyStack.shift();
        } else {
            window.historyIndex++;
        }
        updateHistoryUI();
    };

    window.undo = function() {
        if (window.historyIndex > 0) {
            window.isHistoryNavigating = true;
            window.historyIndex--;
            restoreHistoryState(window.historyStack[window.historyIndex]);
            window.isHistoryNavigating = false;
        }
    };

    window.redo = function() {
        if (window.historyIndex < window.historyStack.length - 1) {
            window.isHistoryNavigating = true;
            window.historyIndex++;
            restoreHistoryState(window.historyStack[window.historyIndex]);
            window.isHistoryNavigating = false;
        }
    };

    function restoreHistoryState(stateString) {
        try {
            editState.resume = JSON.parse(stateString);
            if (typeof window.renderEditorPanels === 'function') window.renderEditorPanels();
            if (typeof window.renderPreview === 'function') window.renderPreview();
            updateHistoryUI();
            if (typeof window.showToast === 'function') window.showToast('状态回滚完成');
            
            // 物理落盘，绕过新历史记录生成
            if (typeof window.saveStore === 'function') {
                const store = window.loadStore();
                const idx = store.resumes.findIndex(r => r.id === editState.resume.id);
                if (idx !== -1) store.resumes[idx] = editState.resume;
                window.saveStore(store);
            }
        } catch (e) {
            console.error('状态回滚异常:', e);
        }
    }

    function updateHistoryUI() {
        const btnUndo = document.getElementById('btn-undo');
        const btnRedo = document.getElementById('btn-redo');
        if (btnUndo) {
            btnUndo.disabled = window.historyIndex <= 0;
            btnUndo.style.opacity = btnUndo.disabled ? '0.5' : '1';
            btnUndo.style.cursor = btnUndo.disabled ? 'not-allowed' : 'pointer';
        }
        if (btnRedo) {
            btnRedo.disabled = window.historyIndex >= window.historyStack.length - 1;
            btnRedo.style.opacity = btnRedo.disabled ? '0.5' : '1';
            btnRedo.style.cursor = btnRedo.disabled ? 'not-allowed' : 'pointer';
        }
    }

    // 拦截 markDirty 算子，挂载防抖机制（500ms）
    const _originalMarkDirty = window.markDirty;
    window.markDirty = function() {
        if (!window.isHistoryNavigating) {
            clearTimeout(window._historyDebounce);
            window._historyDebounce = setTimeout(window.recordHistory, 500);
        }
        if (typeof _originalMarkDirty === 'function') return _originalMarkDirty.apply(this, arguments);
    };

    // 拦截 loadEditor 算子，初始化当前简历快照
    const _originalLoadEditor = window.loadEditor;
    window.loadEditor = function(id) {
        const res = typeof _originalLoadEditor === 'function' ? _originalLoadEditor.apply(this, arguments) : undefined;
        window.historyStack = [];
        window.historyIndex = -1;
        window.isHistoryNavigating = false;
        setTimeout(window.recordHistory, 200);
        return res;
    };

    // 绑定物理键盘事件侦听
    document.addEventListener('keydown', (e) => {
        if ((e.ctrlKey || e.metaKey) && e.key === 'z') {
            if (e.shiftKey) {
                e.preventDefault();
                window.redo();
            } else {
                e.preventDefault();
                window.undo();
            }
        }
    });
})();
</script>
